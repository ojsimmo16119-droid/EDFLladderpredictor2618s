<script>
document.addEventListener("input", updateLadder);

function updateLadder() {
  const games = document.querySelectorAll("[data-game]");
  const ladder = {};

  games.forEach(g => {
    const home = g.children[0].innerText;
    const away = g.children[2].innerText;
    const hs = parseInt(g.querySelector(".home").value);
    const as = parseInt(g.querySelector(".away").value);

    if (!ladder[home]) ladder[home] = {p:0,w:0,l:0,f:0,a:0,pts:0};
    if (!ladder[away]) ladder[away] = {p:0,w:0,l:0,f:0,a:0,pts:0};

    if (isNaN(hs) || isNaN(as)) return;

    ladder[home].p++; ladder[away].p++;
    ladder[home].f += hs; ladder[home].a += as;
    ladder[away].f += as; ladder[away].a += hs;

    if (hs > as) { ladder[home].w++; ladder[home].pts += 4; ladder[away].l++; }
    else if (as > hs) { ladder[away].w++; ladder[away].pts += 4; ladder[home].l++; }
  });

  const table = document.getElementById("ladderTable");
  table.innerHTML = "<tr><th>Team</th><th>P</th><th>W</th><th>L</th><th>Pts</th><th>%</th></tr>";

  Object.entries(ladder)
    .sort((a,b)=>b[1].pts-a[1].pts || ((b[1].f/b[1].a)-(a[1].f/a[1].a)))
    .forEach(([t,s])=>{
      const pct = s.a ? ((s.f/s.a)*100).toFixed(1) : "0.0";
      table.innerHTML += `<tr><td>${t}</td><td>${s.p}</td><td>${s.w}</td><td>${s.l}</td><td>${s.pts}</td><td>${pct}</td></tr>`;
    });
}
